import networkx as nx

graph = nx.Graph()
graph.add_edge('A', 'B', weight=4)
graph.add_edge('A', 'C', weight=2)
graph.add_edge('B', 'C', weight=1)
graph.add_edge('B', 'D', weight=5)
graph.add_edge('C', 'D', weight=8)
graph.add_edge('C', 'E', weight=10)
graph.add_edge('D', 'E', weight=2)
graph.add_edge('D', 'F', weight=6)
graph.add_edge('E', 'F', weight=3)

try:
    shortest_path = nx.shortest_path(graph, source='A', target='F', weight='weight')
    shortest_distance = nx.shortest_path_length(graph, source='A', target='F', weight='weight')
    _result = {
        "status": "OK",
        "query": "Shortest route and distance from A to F",
        "answer": shortest_distance,
        "path": shortest_path
    }
except nx.NetworkXNoPath:
    _result = {
        "status": "OK",
        "query": "Shortest route and distance from A to F",
        "answer": -1,
        "path": []
    }
