import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score
from sklearn.naive_bayes import MultinomialNB

# Load the dataset
df = pd.DataFrame({
    'Message ID': [1, 2, 3, 4],
    'Text': ['My account has been charged twice for the same order #12345', 
             'This is an example message.', 
             '[support_categories(...)]', 
             '[order_context_prompt(...)]']
})

# Define the classification labels
labels = {
    1: 'Error',
    2: 'Support Category',
    3: 'Order Context Prompt'
}

# Split the dataset into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(df['Text'], df['Message ID'], test_size=0.2, random_state=42)

# Create a TF-IDF vectorizer to transform text data into numerical features
vectorizer = TfidfVectorizer()

# Fit the vectorizer to the training data and transform both the training and testing data
X_train_tfidf = vectorizer.fit_transform(X_train)
X_test_tfidf = vectorizer.transform(X_test)

# Train a Multinomial Naive Bayes classifier on the training data
clf = MultinomialNB()
clf.fit(X_train_tfidf, y_train)

# Make predictions on the testing data
y_pred = clf.predict(X_test_tfidf)

# Evaluate the model's accuracy
accuracy = accuracy_score(y_test, y_pred)
print(f'Model Accuracy: {accuracy:.3f}')

# Define a function to detect urgency in new messages
def detect_urgency(message):
    # Transform the message into numerical features using the TF-IDF vectorizer
    message_tfidf = vectorizer.transform([message])
    
    # Make predictions on the transformed message
    predicted_label = clf.predict(message_tfidf)
    
    # Return the predicted label as a string
    return labels[predicted_label[0]]

# Test the function with new messages
new_messages = [
    'My account has been charged twice for the same order #12345',
    '[support_categories(...)]',
    '[order_context_prompt(...)]'
]

for message in new_messages:
    label = detect_urgency(message)
    print(f'Message: {message}, Label: {label}')
