from collections import defaultdict

def normalize_transcript(transcript):
    speakers = defaultdict(list)
    
    # Define a regular expression pattern to match actions
    action_pattern = r'(\w+): (action|fix|update) (.*)'
    
    for line in transcript.split('\n'):
        match = re.match(action_pattern, line)
        
        if match:
            speaker, action_type, *text = match.groups()
            
            # Add the action to the speaker's list
            speakers[speaker].append({'action': action_type, 'text': text[0] if text else ''})
