app = Flask(__name__)

url_db = {

@app.route('/', methods=['GET', 'POST'])
def index():
    if request.method == 'POST':
        long_url = request.form['long_url']
        short_url = str(uuid.uuid4())
        url_db[short_url] = long_url
        redis_client.set(short_url, long_url)
        return redirect(url_for('shorten', short_url=short_url), code=301) # 301 - Permanent Redirect
    return '''
    <form method="post">
        Long URL: <input type="text" name="long_url">
        <button type="submit">Shorten</button>
    </form>
    '''

@app.route('/short/<short_url>')
def shorten(short_url):
    long_url = url_db.get(short_url)
    if long_url:
        return redirect(long_url, code=301)
    else:
        return "URL not found", 404

if __name__ == '__main__':
    app.run(debug=True)
