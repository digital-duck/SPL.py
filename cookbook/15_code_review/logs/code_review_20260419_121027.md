class SeverityScore:
    def __init__(self, name, description, severity):
        """
        Initialize SeverityScore object.

        Args:
            name (str): Name of the language.
            description (str): Description of the vulnerability.
            severity (int): Severity score of the vulnerability (1-5).
        """
        self.name = name
        self.description = description
        self.severity = severity

    def __repr__(self):
        return f"{self.name}: {self.description} (Severity: {self.severity})"


class Vulnerability:
    def __init__(self, language, vulnerability_name, description, severity_score):
        """
        Initialize Vulnerability object.

        Args:
            language (str): Language involved in the vulnerability.
            vulnerability_name (str): Name of the vulnerability.
            description (str): Description of the vulnerability.

        """
        self.language = language
        self.vulnerability_name = vulnerability_name
        self.description = description
        self.severity_score = severity_score

    def __repr__(self):
        return f"{self.language} - {self.vulnerability_name}: {self.description}"

def calculate_severity_score(input_string):
    """
    Calculate the severity score of a given input string.

    The severity score is calculated based on the length, number of vowels and consonants in the string.
    Each vowel counts 1 point, each consonant counts -1 point, and the length of the word counts 2 points for every two characters.

    Args:
        input_string (str): The input string to calculate the severity score.

    Returns:
        int: The calculated severity score.
    """
    vowels = 'aeiou'
    severity_score = 0

    # Calculate the length of the string
    string_length = len(input_string)

    # Iterate over each character in the string
    for char in input_string:
        # Check if the character is a vowel
        if char.lower() in vowels:
            # If it's a vowel, add 1 to the severity score
            severity_score += 1
        elif char.isalpha():  # Check if the character is an alphabet letter
            # If it's an alphabet letter, subtract 1 from the severity score
            severity_score -= 1

    # Calculate the length-based scoring
    string_length_score = (string_length // 2) * 2

    # Add the length-based score to the severity score
    severity_score += string_length_score

    return severity_score
