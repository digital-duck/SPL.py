    """
    Assemble a sentiment report for multiple texts and product reviews.

    Args:
        texts (list): List of text strings to analyze.

    Returns:
        dict: A dictionary with assembled sentiment analysis results.
    """

    # Initialize SentimentIntensityAnalyzer
    sia = SentimentIntensityAnalyzer()

    # Analyze sentiment for each text in batches
    batch_results = []
    for i, review in enumerate(texts):
        # Get the corresponding product review and concatenate reviews
        product_review = product_reviews[i % len(product_reviews)] if i < len(product_reviews) else None

        # Analyze sentiment
        scores = sia.polarity_scores(combined_review)

        # Create a dictionary with sentiment analysis results
        result = {
            'text': review,
            'sentiment': scores['compound'],
            'pos': scores['pos'],
            'neg': scores['neg'],
            'neu': scores['neu']
        }

        # Add result to the batch results list
        batch_results.append(result)

    # Assemble sentiment analysis results into a dictionary
    assembled_report = {
        'text_analysis': [result['text'] for result in batch_results],
        'sentiment_scores': [result['sentiment'], result['pos'], result['neg'], result['neu'] for result in batch_results]
    }

    return assembled_report
