from nltk.sentiment.vader import SentimentIntensityAnalyzer
from collections import defaultdict
import json

def summarize_sentiment_trends(product_reviews):
    """
    Calculate the batch sentiment of a given list of product reviews.

    Parameters:
    product_reviews (list): List of dictionaries representing product reviews with their corresponding metadata.

    Returns:
    A dictionary containing the summary of trends in customer sentiment.
    """

    # Initialize sentiment intensity analyzer
    sia = SentimentIntensityAnalyzer()

    # Initialize dictionary to store sentiment scores for each product category
    sentiment_scores = defaultdict(int)

    # Initialize dictionary to store sentiment counts for each product category
    sentiment_counts = defaultdict(int)

    # Iterate over product reviews and calculate sentiment score
    for review in product_reviews:
        # Calculate sentiment score using VADER
        sentiment_score = sia.polarity_scores(review['text'])['compound']

        # Get product category from metadata
        product_category = review.get('product_category')

        # Increment sentiment scores dictionary with compound score
        if product_category is not None:
            sentiment_scores[product_category] += sentiment_score

        # Increment sentiment counts dictionary with 1 for each review
        sentiment_counts[product_category] += 1

    # Calculate average sentiment scores
    avg_sentiment_scores = {category: score / count for category, (score, count) in sentiment_scores.items()}

    return {
        'avg_sentiment_scores': avg_sentiment_scores,
        'sentiment_counts': dict(sentiment_counts)
    }
