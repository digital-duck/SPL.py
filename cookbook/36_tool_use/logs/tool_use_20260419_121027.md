def generate_sales_report():
    # Input Data
    date = input("Enter Date (H1 2025): ")
    sales_data = list(map(float, input("Enter sales figures (comma-separated): ").split(',')))
    currency1 = input("Enter Currency 1 value: ")
    currency2_value = float(input("Enter Currency 2 value: "))
    currency2_unit = 'USD'
    currency1_value = float(input("Enter Currency 1 value: "))
    currency3_change = float(input("Enter Currency 3 change (as a percentage): ")) / 100

    # Calculate Total Sales
    total_sales = sum(sales_data)

    # Convert Currency 1 and 2 to Decimal
    currency1_value = currency1.replace('$', '').replace(',', '')
    currency2_value = currency2_value.split(' ')[0]

    # Generate Report
    report = f"**{date} Sales Report**\n\n"
    report += f"Total Sales: ${total_sales:.2f}\n"
    report += f"{currency1} Sales: ${currency1_value * total_sales / 100:.2f}\n"
    report += f"{currency2_unit} Sales: ${currency2_value * total_sales / 100:.2f}\n"
    report += f"Currency 3 Change: {currency1_value * currency3_change:.2f}%\n"

    return report

def main():
    print(generate_sales_report())

if __name__ == "__main__":
    main()
